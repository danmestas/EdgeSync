```go
// storeReceivedFile validates and stores a received file/cfile blob.
// If deltaSrc is non-empty and the source is not found, it returns
// ErrDeltaSourceMissing so the caller can handle it (phantom in clone, error in sync).
var ErrDeltaSourceMissing = fmt.Errorf("delta source not found")

func storeReceivedFile(r *repo.Repo, uuid, deltaSrc string, payload []byte) error {
	if r == nil { panic("sync.storeReceivedFile: r must not be nil") }
	if uuid == "" { panic("sync.storeReceivedFile: uuid must not be empty") }
	if payload == nil { panic("sync.storeReceivedFile: payload must not be nil") }
	if !hash.IsValidHash(uuid) {
		return fmt.Errorf("sync: invalid UUID format: %s", uuid)
	}

	var fullContent []byte

	if deltaSrc != "" {
		srcRid, ok := blob.Exists(r.DB(), deltaSrc)
		if !ok {
			return fmt.Errorf("%w: %s", ErrDeltaSourceMissing, deltaSrc)
		}
		baseContent, err := content.Expand(r.DB(), srcRid)
		if err != nil {
			return fmt.Errorf("expanding delta source %s: %w", deltaSrc, err)
		}
		applied, err := delta.Apply(baseContent, payload)
		if err != nil {
			return fmt.Errorf("applying delta for %s: %w", uuid, err)
		}
		fullContent = applied
	} else {
		fullContent = payload
	}

	var computedUUID string
	if len(uuid) > 40 {
		computedUUID = hash.SHA3(fullContent)
	} else {
		computedUUID = hash.SHA1(fullContent)
	}
	if computedUUID != uuid {
		return fmt.Errorf("UUID mismatch for received file: expected %s, got %s", uuid, computedUUID)
	}

	return r.WithTx(func(tx *db.Tx) error {
		if rid, ok := blob.Exists(tx, uuid); ok {
			_, err := tx.Exec("INSERT OR IGNORE INTO unclustered(rid) VALUES(?)", rid)
			return err
		}
		compressed, err := blob.Compress(fullContent)
		if err != nil {
			return err
		}
		result, err := tx.Exec(
			"INSERT INTO blob(uuid, size, content, rcvid) VALUES(?, ?, ?, 1)",
			uuid, len(fullContent), compressed,
		)
		if err != nil {
			return err
		}
		rid, err := result.LastInsertId()
		if err != nil {
			return err
		}
		_, err = tx.Exec("INSERT OR IGNORE INTO unclustered(rid) VALUES(?)", rid)
		return err
	})
}
```

```go
package sync

import (
	"context"
	"errors"
	"fmt"
	"os"

	"github.com/dmestas/edgesync/go-libfossil/blob"
	"github.com/dmestas/edgesync/go-libfossil/repo"
	"github.com/dmestas/edgesync/go-libfossil/simio"
	"github.com/dmestas/edgesync/go-libfossil/xfer"
)

type cloneSession struct {
	repo                *repo.Repo
	env                 *simio.Env
	opts                CloneOpts
	seqno               int
	projectCode         string
	serverCode          string
	cookie              string
	blobsRecvd          int
	phantoms            map[string]bool
	filesRecvdThisRound int
	messages            []string
}

// Clone creates a new repository at path by cloning from a remote Fossil server.
// On success, returns the opened repo and clone result.
// On failure, the partially-created repo file is deleted.
func Clone(ctx context.Context, path string, t Transport, opts CloneOpts) (r *repo.Repo, result *CloneResult, err error) {
	if path == "" { panic("sync.Clone: path must not be empty") }
	if t == nil { panic("sync.Clone: t must not be nil") }

	// Check path doesn't already exist.
	if _, statErr := os.Stat(path); statErr == nil {
		return nil, nil, fmt.Errorf("sync.Clone: file already exists: %s", path)
	}

	env := opts.Env
	if env == nil {
		env = simio.RealEnv()
	}
	version := opts.Version
	if version <= 0 {
		version = 3
	}

	// Create the repo.
	user := opts.User
	if user == "" {
		user = "setup"
	}
	r, err = repo.Create(path, user, env.Rand)
	if err != nil {
		return nil, nil, fmt.Errorf("sync.Clone: create repo: %w", err)
	}

	// Delete project-code — server will provide it.
	if _, err := r.DB().Exec("DELETE FROM config WHERE name='project-code'"); err != nil {
		r.Close()
		os.Remove(path)
		return nil, nil, fmt.Errorf("sync.Clone: clear project-code: %w", err)
	}

	// Run clone loop.
	cs := &cloneSession{
		repo:     r,
		env:      env,
		opts:     opts,
		seqno:    1,
		phantoms: make(map[string]bool),
	}

	result, err = cs.cloneLoop(ctx, t, version)
	if err != nil {
		r.Close()
		os.Remove(path)
		return nil, nil, err
	}

	return r, result, nil
}

func (cs *cloneSession) cloneLoop(ctx context.Context, t Transport, version int) (*CloneResult, error) {
	for cycle := 0; ; cycle++ {
		select {
		case <-ctx.Done():
			return nil, ctx.Err()
		default:
		}
		if cycle >= MaxRounds {
			return nil, fmt.Errorf("sync.Clone: exceeded %d rounds", MaxRounds)
		}

		req := cs.buildCloneRequest(cycle, version)
		resp, err := t.Exchange(ctx, req)
		if err != nil {
			return nil, fmt.Errorf("sync.Clone: exchange round %d: %w", cycle, err)
		}

		if err := cs.processCloneResponse(resp); err != nil {
			return nil, fmt.Errorf("sync.Clone: process round %d: %w", cycle, err)
		}

		// Store project-code and server-code on first receipt.
		if cs.projectCode != "" {
			if _, err := cs.repo.DB().Exec(
				"REPLACE INTO config(name,value,mtime) VALUES('project-code', ?, julianday('now'))",
				cs.projectCode,
			); err != nil {
				return nil, fmt.Errorf("sync.Clone: store project-code: %w", err)
			}
		}
		if cs.serverCode != "" {
			if _, err := cs.repo.DB().Exec(
				"REPLACE INTO config(name,value,mtime) VALUES('server-code', ?, julianday('now'))",
				cs.serverCode,
			); err != nil {
				return nil, fmt.Errorf("sync.Clone: store server-code: %w", err)
			}
		}

		// Convergence check (matching Fossil xfer.c:2995-3004).
		done := cs.checkConvergence(cycle)
		if done {
			return &CloneResult{
				Rounds:      cycle + 1,
				BlobsRecvd:  cs.blobsRecvd,
				ProjectCode: cs.projectCode,
				ServerCode:  cs.serverCode,
			}, nil
		}
	}
}

func (cs *cloneSession) checkConvergence(cycle int) bool {
	// Always go at least 2 rounds.
	if cycle < 1 {
		return false
	}
	// Continue if sequential delivery not finished.
	if cs.seqno > 0 {
		return false
	}
	// Continue if phantoms remain and we received files this round.
	if len(cs.phantoms) > 0 && cs.filesRecvdThisRound > 0 {
		return false
	}
	// Done if seqno==0, no phantoms (or no progress on phantoms).
	return true
}

func (cs *cloneSession) buildCloneRequest(cycle, version int) *xfer.Message {
	var cards []xfer.Card

	// Pragma: client-version.
	cards = append(cards, &xfer.PragmaCard{
		Name:   "client-version",
		Values: []string{"22800", "20260315", "120000"},
	})

	// Clone card.
	cards = append(cards, &xfer.CloneCard{
		Version: version,
		SeqNo:   cs.seqno,
	})

	// Cookie if cached.
	if cs.cookie != "" {
		cards = append(cards, &xfer.CookieCard{Value: cs.cookie})
	}

	// Gimme cards for phantoms — only after seqno reaches 1
	// (matching Fossil xfer.c:2284-2285).
	if cs.seqno <= 1 {
		for uuid := range cs.phantoms {
			cards = append(cards, &xfer.GimmeCard{UUID: uuid})
		}
	}

	msg := &xfer.Message{Cards: cards}

	// Login card — skip round 0, require project-code.
	if cycle > 0 && cs.opts.User != "" && cs.projectCode != "" {
		payload, _ := msg.Encode()
		payload = appendRandomComment(payload, cs.env.Rand)
		loginCard := computeLogin(cs.opts.User, cs.opts.Password, cs.projectCode, payload)
		msg.Cards = append([]xfer.Card{loginCard}, msg.Cards...)
	}

	return msg
}

func (cs *cloneSession) processCloneResponse(msg *xfer.Message) error {
	if msg == nil { panic("sync.cloneSession.processCloneResponse: msg must not be nil") }

	cs.filesRecvdThisRound = 0

	for _, card := range msg.Cards {
		switch c := card.(type) {
		case *xfer.PushCard:
			// Server provides project-code and server-code.
			if cs.projectCode == "" {
				cs.projectCode = c.ProjectCode
			}
			if cs.serverCode == "" {
				cs.serverCode = c.ServerCode
			}

		case *xfer.FileCard:
			err := storeReceivedFile(cs.repo, c.UUID, c.DeltaSrc, c.Content)
			if err != nil {
				if errors.Is(err, ErrDeltaSourceMissing) {
					cs.phantoms[c.UUID] = true
					// Store as phantom so we can request it later.
					blob.StorePhantom(cs.repo.DB(), c.UUID)
					continue
				}
				return err
			}
			cs.blobsRecvd++
			cs.filesRecvdThisRound++
			delete(cs.phantoms, c.UUID)

		case *xfer.CFileCard:
			err := storeReceivedFile(cs.repo, c.UUID, c.DeltaSrc, c.Content)
			if err != nil {
				if errors.Is(err, ErrDeltaSourceMissing) {
					cs.phantoms[c.UUID] = true
					blob.StorePhantom(cs.repo.DB(), c.UUID)
					continue
				}
				return err
			}
			cs.blobsRecvd++
			cs.filesRecvdThisRound++
			delete(cs.phantoms, c.UUID)

		case *xfer.CloneSeqNoCard:
			cs.seqno = c.SeqNo

		case *xfer.CookieCard:
			cs.cookie = c.Value

		case *xfer.ErrorCard:
			return fmt.Errorf("server error: %s", c.Message)

		case *xfer.MessageCard:
			cs.messages = append(cs.messages, c.Message)

		// Ignore other card types (pragma, etc.)
		}
	}

	return nil
}
```

```go
package main

import (
	"context"
	"fmt"
	"time"

	"github.com/dmestas/edgesync/go-libfossil/sync"
)

type RepoCloneCmd struct {
	URL  string `arg:"" help:"Remote Fossil server URL"`
	Path string `arg:"" help:"Local path for new repository file"`
	User string `short:"u" help:"Username for clone auth"`
	Pass string `short:"p" help:"Password for clone auth"`
}

func (c *RepoCloneCmd) Run(g *Globals) error {
	ctx, cancel := context.WithTimeout(context.Background(), 10*time.Minute)
	defer cancel()

	transport := &sync.HTTPTransport{URL: c.URL}
	opts := sync.CloneOpts{
		User:     c.User,
		Password: c.Pass,
	}

	r, result, err := sync.Clone(ctx, c.Path, transport, opts)
	if err != nil {
		return fmt.Errorf("clone failed: %w", err)
	}
	defer r.Close()

	fmt.Printf("Cloned into %s\n", c.Path)
	fmt.Printf("  Rounds:       %d\n", result.Rounds)
	fmt.Printf("  Blobs:        %d\n", result.BlobsRecvd)
	fmt.Printf("  Project-code: %s\n", result.ProjectCode)
	return nil
}
```
