# WASM Leaf Agent Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Compile the leaf agent to WASI and browser WASM — same sync protocol, two targets.

**Architecture:** Add Storage interface to simio/ (following Clock/Rand pattern), wire it through repo and clone, build-tag browser-specific code (OPFS SQLite, NATS WebSocket dialer, no-op ServeHTTP), provide JS entry point with Web Worker wrapper.

**Tech Stack:** Go 1.23, GOOS=wasip1/js, ncruces SQLite (wazero-backed), NATS WebSocket, syscall/js for browser APIs.

**Spec:** `docs/superpowers/specs/2026-03-20-wasm-leaf-design.md`

```go
// storage.go
package simio

import "os"

// Storage abstracts filesystem operations for WASM portability.
// OSStorage is the default (real OS). MemStorage is for tests and browser fallback.
type Storage interface {
    Stat(path string) (os.FileInfo, error)
    Remove(path string) error
    MkdirAll(path string, perm os.FileMode) error
    ReadFile(path string) ([]byte, error)
    WriteFile(path string, data []byte, perm os.FileMode) error
}

// OSStorage delegates to the os package. Used in production and WASI.
type OSStorage struct{}

func (OSStorage) Stat(path string) (os.FileInfo, error)                       { return os.Stat(path) }
func (OSStorage) Remove(path string) error                                     { return os.Remove(path) }
func (OSStorage) MkdirAll(path string, perm os.FileMode) error                { return os.MkdirAll(path, perm) }
func (OSStorage) ReadFile(path string) ([]byte, error)                         { return os.ReadFile(path) }
func (OSStorage) WriteFile(path string, data []byte, perm os.FileMode) error  { return os.WriteFile(path, data, perm) }
```

```go
// storage_mem.go
package simio

import (
    "fmt"
    "os"
    "strings"
    "time"
)

// MemStorage is an in-memory Storage for tests and browser fallback.
type MemStorage struct {
    files map[string][]byte
    dirs  map[string]bool
}

func NewMemStorage() *MemStorage {
    return &MemStorage{
        files: make(map[string][]byte),
        dirs:  make(map[string]bool),
    }
}

func (m *MemStorage) Stat(path string) (os.FileInfo, error) {
    if m.dirs[path] {
        return &memFileInfo{name: path, isDir: true}, nil
    }
    data, ok := m.files[path]
    if !ok {
        return nil, fmt.Errorf("stat %s: %w", path, os.ErrNotExist)
    }
    return &memFileInfo{name: path, size: int64(len(data))}, nil
}

func (m *MemStorage) Remove(path string) error {
    if _, ok := m.files[path]; !ok {
        if !m.dirs[path] {
            return fmt.Errorf("remove %s: %w", path, os.ErrNotExist)
        }
    }
    delete(m.files, path)
    delete(m.dirs, path)
    return nil
}

func (m *MemStorage) MkdirAll(path string, _ os.FileMode) error {
    m.dirs[path] = true
    // Create parent dirs
    for path != "/" && path != "." && path != "" {
        path = path[:strings.LastIndex(path, "/")]
        if path == "" {
            path = "/"
        }
        m.dirs[path] = true
        if path == "/" {
            break
        }
    }
    return nil
}

func (m *MemStorage) ReadFile(path string) ([]byte, error) {
    data, ok := m.files[path]
    if !ok {
        return nil, fmt.Errorf("read %s: %w", path, os.ErrNotExist)
    }
    cp := make([]byte, len(data))
    copy(cp, data)
    return cp, nil
}

func (m *MemStorage) WriteFile(path string, data []byte, _ os.FileMode) error {
    cp := make([]byte, len(data))
    copy(cp, data)
    m.files[path] = cp
    return nil
}

// memFileInfo implements os.FileInfo for MemStorage.
type memFileInfo struct {
    name  string
    size  int64
    isDir bool
}

func (f *memFileInfo) Name() string      { return f.name }
func (f *memFileInfo) Size() int64       { return f.size }
func (f *memFileInfo) Mode() os.FileMode { if f.isDir { return os.ModeDir | 0755 }; return 0644 }
func (f *memFileInfo) ModTime() time.Time { return time.Time{} }
func (f *memFileInfo) IsDir() bool       { return f.isDir }
func (f *memFileInfo) Sys() interface{}  { return nil }
```

```go
//go:build js

package main

import (
    "context"
    "syscall/js"

    "github.com/dmestas/edgesync/leaf/agent"
)

var (
    currentAgent *agent.Agent
    ctx          context.Context
    cancel       context.CancelFunc
)

func jsNewAgent(_ js.Value, args []js.Value) interface{} {
    if len(args) < 1 {
        return js.ValueOf("error: config object required")
    }
    cfg := args[0]
    // Parse JS object into agent.Config
    config := agent.Config{
        RepoPath:      cfg.Get("repoPath").String(),
        NATSUrl:       cfg.Get("natsUrl").String(),
        Push:          true,
        Pull:          true,
        SubjectPrefix: "fossil",
    }
    var err error
    currentAgent, err = agent.New(config)
    if err != nil {
        return js.ValueOf("error: " + err.Error())
    }
    return js.ValueOf("ok")
}

func jsStart(_ js.Value, _ []js.Value) interface{} {
    if currentAgent == nil {
        return js.ValueOf("error: no agent")
    }
    ctx, cancel = context.WithCancel(context.Background())
    if err := currentAgent.Start(); err != nil {
        return js.ValueOf("error: " + err.Error())
    }
    return js.ValueOf("ok")
}

func jsStop(_ js.Value, _ []js.Value) interface{} {
    if currentAgent != nil {
        currentAgent.Stop()
    }
    if cancel != nil {
        cancel()
    }
    return js.ValueOf("ok")
}

func jsSyncNow(_ js.Value, _ []js.Value) interface{} {
    if currentAgent != nil {
        currentAgent.SyncNow()
    }
    return js.ValueOf("ok")
}

func main() {
    js.Global().Set("edgesync", js.ValueOf(map[string]interface{}{
        "newAgent": js.FuncOf(jsNewAgent),
        "start":    js.FuncOf(jsStart),
        "stop":     js.FuncOf(jsStop),
        "syncNow":  js.FuncOf(jsSyncNow),
    }))
    // Block forever — WASM module stays alive
    select {}
}
```

```js
// worker.js — Web Worker wrapper for EdgeSync WASM leaf agent
importScripts("wasm_exec.js"); // Go's WASM support JS (from GOROOT/misc/wasm/)

let go;

self.onmessage = async function(e) {
    const msg = e.data;

    switch (msg.type) {
        case "init":
            go = new Go();
            const result = await WebAssembly.instantiateStreaming(
                fetch(msg.wasmUrl || "leaf-browser.wasm"),
                go.importObject
            );
            go.run(result.instance);
            // Agent is now available via edgesync global
            const initResult = edgesync.newAgent(msg.config || {});
            self.postMessage({ type: "init", result: initResult });
            break;

        case "start":
            const startResult = edgesync.start();
            self.postMessage({ type: "started", result: startResult });
            break;

        case "sync":
            const syncResult = edgesync.syncNow();
            self.postMessage({ type: "synced", result: syncResult });
            break;

        case "stop":
            const stopResult = edgesync.stop();
            self.postMessage({ type: "stopped", result: stopResult });
            break;
    }
};
```
