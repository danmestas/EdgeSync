```go
// TransportFunc adapts a plain function to the Transport interface.
// This is the Transport equivalent of http.HandlerFunc.
type TransportFunc func(ctx context.Context, payload []byte) ([]byte, error)

// RoundTrip calls the function.
func (f TransportFunc) RoundTrip(ctx context.Context, payload []byte) ([]byte, error) {
	return f(ctx, payload)
}
```

```go
// runCloneDST performs a clone through MockFossil with Buggify enabled
// on both client and server. Verifies convergence or clean failure.
func runCloneDST(t *testing.T, seed int64) {
	t.Helper()

	// Create and seed server repo.
	serverPath := filepath.Join(t.TempDir(), "server.fossil")
	serverRepo, err := repo.Create(serverPath, "server", simio.CryptoRand{})
	if err != nil {
		t.Fatalf("repo.Create server: %v", err)
	}
	defer serverRepo.Close()

	// Seed 50 blobs with deterministic content.
	rng := rand.New(rand.NewSource(seed))
	seededUUIDs := make([]string, 50)
	for i := range 50 {
		data := make([]byte, 100+rng.Intn(900))
		rng.Read(data)
		rid, uuid, err := blob.Store(serverRepo.DB(), data)
		if err != nil {
			t.Fatalf("blob.Store %d: %v", i, err)
		}
		_ = rid
		seededUUIDs[i] = uuid
	}

	// Server-side Buggify.
	mf := NewMockFossil(serverRepo)
	mf.SetBuggify(&SeededBuggify{rng: rand.New(rand.NewSource(seed + 1000))})

	// Client-side Buggify.
	clientBuggify := &SeededBuggify{rng: rand.New(rand.NewSource(seed + 2000))}

	clonePath := filepath.Join(t.TempDir(), "clone.fossil")
	cloneRepo, result, err := libsync.Clone(
		context.Background(),
		clonePath,
		mf, // MockFossil implements sync.Transport
		libsync.CloneOpts{
			Buggify: clientBuggify,
		},
	)

	if err != nil {
		// Expected failure path (corruptHash or persistent auth failure).
		// Verify cleanup: repo file should be deleted.
		if _, statErr := os.Stat(clonePath); !os.IsNotExist(statErr) {
			t.Errorf("seed %d: clone failed but repo file still exists: %v", seed, err)
		}
		t.Logf("seed %d: clone failed (expected under faults): %v", seed, err)
		return
	}
	defer cloneRepo.Close()

	t.Logf("seed %d: clone succeeded in %d rounds, %d blobs received",
		seed, result.Rounds, result.BlobsRecvd)

	// Invariant 1: blob convergence.
	for _, uuid := range seededUUIDs {
		if _, exists := blob.Exists(cloneRepo.DB(), uuid); !exists {
			t.Errorf("seed %d: clone missing blob %s", seed, uuid)
		}
	}

	// Invariant 2: no phantoms remain.
	var phantomCount int
	cloneRepo.DB().QueryRow("SELECT COUNT(*) FROM phantom").Scan(&phantomCount)
	if phantomCount > 0 {
		t.Errorf("seed %d: clone has %d phantoms remaining", seed, phantomCount)
	}
}

func TestCloneDST(t *testing.T) {
	runCloneDST(t, 42)
}

func TestCloneDSTSeedSweep(t *testing.T) {
	for seed := int64(0); seed < 20; seed++ {
		t.Run(fmt.Sprintf("seed-%d", seed), func(t *testing.T) {
			runCloneDST(t, seed)
		})
	}
}
```
